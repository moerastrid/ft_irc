tokens:
	<name_prefix>:		":nick!user@host"
	<server_prefix>:	":server_hostname"
	<numeric>: 			001 through 999 (always 3 digits)
	<caller>:			The nickname of the function caller.
	<command>:			the command to run (in CAPITALS)


At global level:
ERR_UNKNOWNCOMMAND:
	<server_prefix> <numeric> <caller> <command> "Unknown command from client"
ERR_NEEDMOREPARAMS:
	<server_prefix> <numeric> <caller> <command> "Not enough parameters"
ERR_TOOMANYPARAMS:
	<server_prefix> <numeric> <caller> <command> "Too many parameters"

run_pass:
ERR_ALREADYREGISTERED:
	<server_prefix> <numeric> <caller> "You may not reregister"
ERR_PASSWDMISMATCH:
	<server_prefix> <numeric> <caller> "Password incorrect"
